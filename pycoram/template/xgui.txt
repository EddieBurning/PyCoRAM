#Definitional proc to organize widgets for parameters.
proc create_gui { ipview } {
	set Page0 [ ipgui::add_page $ipview  -name "Page 0" -layout vertical]
	set Component_Name [ ipgui::add_param  $ipview  -parent  $Page0  -name Component_Name ]
	set C_M_AXI_DATA_WIDTH [ipgui::add_param $ipview -parent $Page0 -name C_M_AXI_DATA_WIDTH]
	set C_BURST_LEN [ipgui::add_param $ipview -parent $Page0 -name C_BURST_LEN]
	set C_TRANSACTIONS_NUM [ipgui::add_param $ipview -parent $Page0 -name C_TRANSACTIONS_NUM]
}
